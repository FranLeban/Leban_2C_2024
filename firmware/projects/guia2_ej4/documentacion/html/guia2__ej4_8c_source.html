201static void DAC_Conversion(void *pvParameter)//DAC: Digital Analogic Converter
202{
203 while (true)
204 {
205 ulTaskNotifyTake(pdTRUE, portMAX_DELAY);
206 /*La tarea espera en este punto hasta recibir una notificación*/
207
208 //Envío los valores del arreglo ecg a la función AnalogOutputWrite en un bucle, actualizando
209 //el contador para avanzar a través del arreglo. Cuando se llega al final del arreglo, el contador
210 //se restablece a 0 para comenzar de nuevo.
211 AnalogOutputWrite(ecg[contador]);
212 if (contador < BUFFER_SIZE - 1)
213 {
214 contador++;//Se incrementa el valor de contador en 1. Esto hace que el siguiente valor del
215 //arreglo ecg sea enviado a la función AnalogOutputWrite en la próxima iteración.
216 }
217 else
218 {
219 contador = 0;//Se restablece el valor de contador a 0. Esto hace que el bucle comience de
220 //nuevo desde el principio del arreglo ecg.
221 }
222 }
223}
